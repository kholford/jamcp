// Care Plan — Build Single-Page HTML // Compiles multiple care-plan drafts into one standalone HTML document // with an internal table of contents. // // Assumptions: // 1. Each source draft exists in Drafts with the title shown below. // 2. The draft title is the first markdown H1 line, e.g. // # Care Plan Appendix A — Financial Model // 3. The source drafts may include DOC FORMAT comments and centered title divs. // This script strips those wrapper elements and compiles the substantive content. // // Output: // - Creates or updates a draft titled: // Judith A McMahon — Care Plan — Full HTML // // Notes: // - Update SOURCE_DOCS if you rename any documents. // - If one or more source drafts are missing, the output draft will still be built // and will include a "Missing Source Documents" section at the top. const OUTPUT_TITLE = "Judith A McMahon — Care Plan — Full HTML"; const SOURCE_DOCS = [ { title: "Care Plan — Strategy, Approach, and Rationale", navTitle: "Main Document", slug: "main-document" }, { title: "Care Plan Appendix A — Financial Model", navTitle: "Appendix A — Financial Model", slug: "appendix-a" }, { title: "Care Plan Appendix B — Trust Design & Intent", navTitle: "Appendix B — Trust Design & Intent", slug: "appendix-b" }, { title: "Care Plan Appendix C — Financial Admission Narrative (Assisted Living Placement)", navTitle: "Appendix C — Financial Admission Narrative", slug: "appendix-c" }, { title: "Care Plan Appendix D — Timing Flexibility & Transition Risk", navTitle: "Appendix D — Timing Flexibility & Transition Risk", slug: "appendix-d" }, { title: "Care Plan Appendix E — Timeline, Care, Asset Flow & Funding Phases", navTitle: "Appendix E — Timeline, Care, Asset Flow & Funding Phases", slug: "appendix-e" } ]; // ---------- Helpers ---------- function escapeHtml(str) { return String(str) .replace(/&/g, "&") .replace(//g, ">") .replace(/"/g, """); } function getAllDrafts() { // Query all accessible drafts. This is the broadest, simplest approach. // If you prefer to restrict by workspace or tag, adjust here. return Draft.query("", "all") || []; } function findDraftByTitle(title, drafts) { for (let d of drafts) { if (d.title && d.title.trim() === title.trim()) { return d; } } return null; } function stripFrontMatterAndWrapper(source) { let text = String(source || "").replace(/\r\n/g, "\n"); // Remove top H1 title line text = text.replace(/^# .+\n+/, ""); // Remove DOC FORMAT block text = text.replace( /[\s\S]*?\n*/m, "" ); // Remove backlink comment at top, if present text = text.replace(/^\n*/m, ""); // Remove leading centered title div blocks at top (usually one or two) text = text.replace( /^(?:\s*\s*){1,3}/m, "" ); // Trim leading/trailing whitespace return text.trim(); } function renderMarkdownToHtml(markdownText) { const md = Markdown.create(); return md.render(markdownText); } function buildStyles() { return ` :root { --text: #111; --muted: #555; --border: #d9d9d9; --link: #0645ad; --bg: #ffffff; } html, body { margin: 0; padding: 0; background: var(--bg); color: var(--text); font-family: Arial, Helvetica, sans-serif; line-height: 1.45; font-size: 16px; } body { margin: 0.75in; } .page-title { text-align: center; font-size: 1.7em; font-weight: 700; margin: 0 0 0.15em 0; } .page-subtitle { text-align: center; font-size: 1.35em; font-weight: 700; margin: 0 0 1.2em 0; } .toc { border: 1px solid var(--border); padding: 1em 1.1em; margin: 0 0 1.5em 0; } .toc-title { font-weight: 700; margin: 0 0 0.75em 0; font-size: 1.05em; } .toc-group { margin: 0 0 1em 0; } .toc-group-title { font-weight: 700; margin: 0.7em 0 0.45em 0; padding-bottom: 0.15em; border-bottom: 1px solid var(--border); } .toc a { display: block; margin: 0.45em 0; color: var(--link); text-decoration: none; } .toc a:hover, .toc a:focus { text-decoration: underline; } .missing { border: 1px solid #d9b7b7; background: #fff7f7; padding: 0.85em 1em; margin: 0 0 1.5em 0; } .missing h2 { margin-top: 0; font-size: 1.05em; } .document { margin: 0 0 2.2em 0; page-break-before: always; } .document.first-doc { page-break-before: auto; } .document-title { font-size: 1.35em; font-weight: 700; margin: 0 0 0.85em 0; padding-bottom: 0.2em; border-bottom: 2px solid var(--border); } .back-to-top { margin: 1.25em 0 0 0; font-size: 0.95em; } .back-to-top a { color: var(--link); text-decoration: none; } .back-to-top a:hover, .back-to-top a:focus { text-decoration: underline; } p, ul, ol, table, blockquote { margin-top: 0.6em; margin-bottom: 0.6em; } h1, h2, h3, h4 { margin-top: 1em; margin-bottom: 0.4em; } table { border-collapse: collapse; width: 100%; } th, td { border: 1px solid var(--border); padding: 0.45em 0.5em; text-align: left; vertical-align: top; } code, pre { font-family: Menlo, Consolas, monospace; font-size: 0.95em; } pre { white-space: pre-wrap; word-break: break-word; border: 1px solid var(--border); padding: 0.75em; overflow: auto; } @media print { body { margin: 0.5in 0.75in; } .toc, .missing, .back-to-top { break-inside: avoid; } } `; } function buildTocHtml() { const mainDoc = SOURCE_DOCS[0]; const appendices = SOURCE_DOCS.slice(1); let html = `
`; html += `
Table of Contents
`; html += `
`; html += `
Main Document
`; html += `${escapeHtml(mainDoc.navTitle)}`; html += `
`; html += `
`; html += `
Appendices
`; for (const doc of appendices) { html += `${escapeHtml(doc.navTitle)}`; } html += `
`; html += `
`; return html; } function buildMissingHtml(missingTitles) { if (!missingTitles.length) return ""; let html = `
`; html += `

Missing Source Documents

`; html += `

The following source drafts were not found and were not included in the compiled file:

`; html += `
    `; for (const title of missingTitles) { html += `
  • ${escapeHtml(title)}
    • `; } html += `
`; html += `
`; return html; } function buildDocumentSectionHtml(doc, renderedHtml, isFirstDoc) { return `
${escapeHtml(doc.navTitle)}
${renderedHtml}
Back to Table of Contents
`; } function buildFullHtml(sectionsHtml, missingHtml) { return ` ${escapeHtml(OUTPUT_TITLE)}
Judith A McMahon — Care Plan
Compiled Full HTML
${buildTocHtml()} ${missingHtml} ${sectionsHtml.join("\n")} `; } // ---------- Main ---------- const drafts = getAllDrafts(); const sectionsHtml = []; const missing = []; for (let i = 0; i < SOURCE_DOCS.length; i++) { const doc = SOURCE_DOCS[i]; const sourceDraft = findDraftByTitle(doc.title, drafts); if (!sourceDraft) { missing.push(doc.title); continue; } const cleaned = stripFrontMatterAndWrapper(sourceDraft.content); const rendered = renderMarkdownToHtml(cleaned); sectionsHtml.push(buildDocumentSectionHtml(doc, rendered, sectionsHtml.length === 0)); } const finalHtml = buildFullHtml(sectionsHtml, buildMissingHtml(missing)); // Create or update output draft let outputDraft = findDraftByTitle(OUTPUT_TITLE, drafts); if (!outputDraft) { outputDraft = Draft.create(); } outputDraft.content = finalHtml; outputDraft.update(); // Load the compiled draft editor.load(outputDraft); app.displaySuccessMessage("Compiled full HTML draft created/updated.");
